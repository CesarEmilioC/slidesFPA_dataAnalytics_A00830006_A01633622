% ------------------------------ Motivation -----------------------------------
\begin{frame}{Motivation}
\small
\begin{columns}[T,onlytextwidth]
\column{0.58\textwidth}
\begin{itemize}\setlength\itemsep{3pt}
  \item The \heading{S\&P 500} is the most-watched benchmark of U.S. equity health, with 11 GICS sectors that differ in volatility and news flow.
  \item Predicting next-day price \heading{direction} (not price) has direct value for risk management and tactical allocation.
  \item Markets are not perfectly efficient: behavioural biases and delayed information assimilation create exploitable signals.
  \item Recent NLP advances (FinBERT) make it cheap to extract structured sentiment from financial text.
\end{itemize}

\column{0.40\textwidth}
\begin{block}{Research question}
\footnotesize Does a tree-based ensemble of technical indicators \alert{plus} short-form news sentiment beat a technical-only baseline at predicting next-day S\&P 500 direction --- and does the gain grow with sector volatility?
\end{block}
\end{columns}
\end{frame}

% ------------------------------ Problem framing ------------------------------
\begin{frame}{Problem statement and hypotheses}
\small
\begin{block}{Binary classification target}
\centering\small
$y_t = \begin{cases} 1 & \text{if } C_{t+1} > C_t \text{ (upward)} \\ 0 & \text{if } C_{t+1} \leq C_t \text{ (no change or downward)} \end{cases}$
\end{block}

\vspace{2pt}
\begin{columns}[T,onlytextwidth]
\column{0.5\textwidth}
\heading{H1.} Adding FinBERT-derived sentiment to a technical-only baseline \textit{improves} Accuracy, F1, and AUC-ROC.

\vspace{4pt}
\heading{H2.} The sentiment gain is \textit{larger} for sectors with higher historical volatility.

\column{0.5\textwidth}
\heading{Research questions}
\begin{enumerate}\setlength\itemsep{1pt}
  \item Does sentiment improve daily direction prediction?
  \item Random Forest or XGBoost --- which wins?
  \item Does sector volatility modulate the sentiment benefit?
\end{enumerate}
\end{columns}

\vspace{4pt}
\footnotesize \textbf{Note.} Flat days ($C_{t+1}=C_t$) are grouped with class~0; for liquid indices they are rare.
\end{frame}

% ------------------------------ Related Work ---------------------------------
\begin{frame}{Related work --- five anchor references}
\small
\begin{itemize}
  \item \heading{Bollen et al. (2011)} --- Twitter mood predicts DJIA direction with 87.6\% accuracy. \textit{Sentiment carries predictive information beyond price.}
  \item \heading{Vargas et al. (2017)} --- CNN/RNN on Reuters news + technicals beats single-modality baselines. \textit{Multimodal helps, but is compute heavy.}
  \item \heading{Araci (2019)} --- FinBERT, BERT fine-tuned on financial text, sets the SOTA for short-form financial sentiment. \textit{Available pre-trained, drop-in for our pipeline.}
  \item \heading{Zhang et al. (2019)} --- XGBoost with sentiment is competitive with deep models, more interpretable. \textit{Justifies our tree-ensemble choice.}
  \item \heading{L\'opez de Prado (2018)} --- Chronological splits and look-ahead-bias controls are mandatory. \textit{Our experimental design follows this guidance.}
\end{itemize}
\end{frame}

\column{0.48\textwidth}

\heading{Per headline}

\vspace{-4pt}

  \item Batch 32; score = label sign $\times$ confidence.
  \item No headlines $\Rightarrow$ neutral ($S_t = 0$).
\end{itemize}
\end{columns}

% ------------------------------ Train / metrics ------------------------------
\begin{frame}{Training protocol and evaluation}
\small
\begin{columns}[T,onlytextwidth]
\column{0.47\textwidth}
\heading{Chronological 80/20 split}
\begin{itemize}\setlength\itemsep{2pt}
  \item No shuffling --- prevents look-ahead bias.
  \item For each dataset: $\approx 2{,}000$ train days / 501 test days.
  \item Same split for every algorithm and feature configuration.
\end{itemize}

\vspace{4pt}
\heading{Hyper-parameters}
\begin{itemize}\setlength\itemsep{2pt}
  \item RF: \texttt{n\_estimators=100}, default depth, \texttt{random\_state=42}.
  \item XGB: \texttt{n\_estimators=100}, default depth, \texttt{eval\_metric="logloss"}.
\end{itemize}

\column{0.06\textwidth}\,
\column{0.47\textwidth}
\heading{Metrics}
\begin{itemize}\setlength\itemsep{2pt}
  \item \textbf{Accuracy} --- overall correctness.
  \item \textbf{F1-score} --- harmonic mean of P / R, robust to imbalance.
  \item \textbf{AUC-ROC} --- threshold-free ranking quality.
\end{itemize}

\vspace{4pt}
\heading{Why all three}
A trivial ``always up'' classifier can score $\sim 0.55$ Acc and $\sim 0.71$ F1 yet have AUC $\approx 0.50$. Reporting AUC alongside protects against false confidence.
\end{columns}
\end{frame}

% ------------------------------ Hypothesis check -----------------------------
\begin{frame}{Hypothesis assessment}
\small
\begin{block}{H1 --- ``Sentiment improves prediction.''}
\textit{Partially supported.} The broad SP500 index sees a meaningful Random Forest improvement on all three metrics. Sector-level effects are inconsistent: XLE benefits, XLP and XLK do not.
\end{block}

\begin{block}{H2 --- ``Higher volatility $\Rightarrow$ larger sentiment gain.''}
\textit{Partially supported.} XLE (highest vol) shows the largest positive XGB delta. However, XLK (median vol) does not benefit, so the volatility$\rightarrow$benefit gradient is not monotonic.
\end{block}

\begin{block}{RQ2 --- ``Random Forest or XGBoost?''}
On AUC the two algorithms are tied. \textbf{Random Forest} wins on the SP500 index when sentiment is included; \textbf{XGBoost} wins on the high-volatility XLE sector.
\end{block}
\end{frame}

% ------------------------------ Discussion -----------------------------------
\begin{frame}{Discussion}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.50\textwidth}
\begin{itemize}\setlength\itemsep{4pt}
  \item \heading{AUC near 0.50.} Consistent with the weak form of the Efficient Market Hypothesis --- price-only signals offer at most a marginal next-day edge.
  \item \heading{Sentiment is a thin signal at this granularity.} Daily-aggregated headline sentiment loses intra-day timing; noisy sectors mask its contribution.
  \item \heading{Class collapse on XLK.} 2018--2019 had more flat/down than up days; without re-weighting the model picks the majority class.
  \item \heading{Why SP500 benefits the most.} Index returns aggregate idiosyncratic noise, leaving systematic news flow as a cleaner signal.
\end{itemize}

% ------------------------------ Limitations ----------------------------------
\begin{frame}{Limitations}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.55\textwidth}
\begin{itemize}\setlength\itemsep{2pt}
  \item \heading{No statistical significance test.} The $+0.04$ AUC delta is a point estimate; a DeLong test or bootstrap CI on the 501-day test set is needed.
  \item \heading{Transaction-cost reality check} (right). Even at $+0.05$ accuracy, a 5--10\,bp round-trip cost erodes most of the edge --- our results are \textit{scientific}, not yet \textit{tradable}.
  \item \heading{News coverage skew.} Kaggle corpus emphasises U.S.\ mega-caps; smaller sector constituents under-represented.
  \item \heading{Headline-only sentiment.} Body text and analyst-report tone ignored; FinBERT runs on short titles only.
  \item \heading{Sampling cap.} 50 headlines/day is a compute compromise; heavy-news days may be undercounted.
  \item \heading{Single chronological split.} A walk-forward backtest would give stronger evidence than one 80/20 cut.
  \item \heading{Hyper-parameters not tuned.} Defaults are used so the 16 runs stay directly comparable.
\end{itemize}

% ------------------------------ Conclusions ----------------------------------
\begin{frame}{Conclusions}
\small
\begin{block}{Take-aways}
\begin{enumerate}\setlength\itemsep{3pt}
  \item Headline sentiment from FinBERT \alert{helps at the index level} when paired with Random Forest: $+0.05$ Acc, $+0.06$ F1, $+0.04$ AUC over the technical-only baseline --- small but consistent.
  \item The sentiment effect is \alert{not monotonic in sector volatility}: high-vol XLE benefits, low-vol XLP does not, median-vol XLK also fails to benefit. \textbf{H2} is only partially supported.
  \item The absolute AUC $\approx 0.50$ ceiling on individual sectors is \alert{consistent with the weak form of EMH} --- the contribution is the \textit{sign and direction} of the delta, not a tradable signal.
\end{enumerate}
\end{block}

\vspace{2pt}
\begin{center}\footnotesize\itshape\color{TecGray}
Honest framing: the model is \emph{slightly} better than ``always up''; it confirms sentiment carries information but does not break the daily-resolution ceiling.
\end{center}
\end{frame}

% ------------------------------ Future work ----------------------------------
\begin{frame}{Future work}
\footnotesize
\begin{itemize}\setlength\itemsep{2pt}
  \item \heading{Statistical significance of the $\Delta$AUC.} DeLong test and bootstrap confidence intervals on the 501-day test set to confirm sentiment gains are not noise.
  \item \heading{Transaction-cost-aware backtest.} Score the SP500+RF+Sent.\ classifier as a daily long/flat signal net of 5--10\,bp round-trip costs.
  \item \heading{Walk-forward evaluation} with expanding training windows.
  \item \heading{Multi-resolution sentiment.} Pair daily aggregates with $K$-day moving averages and event-time deltas.
  \item \heading{Sector-aware news routing.} Weight headlines by the ticker's sector weight to better align signal with target.
  \item \heading{Class re-weighting or stratification} to fix XLK's collapse onto the majority class.
  \item \heading{Out-of-sample 2020--2024 stress test} once a COVID-normalisation feature is added.
\end{itemize}
\end{frame}

\begin{frame}{Appendix --- Detailed metric definitions}
\footnotesize
\heading{Confusion-matrix counts.} TP, TN, FP, FN are computed on the chronological 20\% test set.

\vspace{4pt}
\begin{center}
\renewcommand{\arraystretch}{1.25}
\begin{tabular}{@{}p{2.0cm} p{4.5cm} p{7.0cm}@{}}
\toprule
\textbf{Metric} & \textbf{Formula} & \textbf{Reading} \\
\midrule
Accuracy   & $\dfrac{TP+TN}{TP+TN+FP+FN}$                          & overall correctness, threshold-dependent \\
Precision  & $\dfrac{TP}{TP+FP}$                                   & of predicted up days, how many were right \\
Recall     & $\dfrac{TP}{TP+FN}$                                   & of actual up days, how many we caught \\
F1-score   & $\dfrac{2\,P\,R}{P+R}$                                & harmonic mean of $P$ and $R$ \\
AUC-ROC    & $\int_0^1 \mathrm{TPR}(\mathrm{FPR}^{-1}(u))\, du$    & threshold-free ranking quality \\
\bottomrule
\end{tabular}
\end{center}

\vspace{4pt}
\heading{Why all three.} A na\"ive ``always-up'' classifier on SP500 (54.7\% up days) gets Acc $0.55$, F1 $\approx 0.71$, yet AUC $= 0.50$.
